# Functions defining initial and boundary conditions
def fun_u_0(x):
    n = x.shape[0]
    return tf.zeros((n,1),dtype=DTYPE)

def fun_u_bc_1(t, x):
    n = t.shape[0]
    return -tf.ones((n, 1), dtype=DTYPE)

def fun_u_bc_2(t, x):
    n = t.shape[0]
    return tf.zeros((n, 1), dtype=DTYPE)

# Define residual of the PDE
def fun_r(t, x, u, u_t, u_x, u_xx):
    return u_t - u_xx

# function for Neumann boundary conditions
def N_bc(model, X):
    with tf.GradientTape(persistent=True) as tape:
        t, x = X[:, 0:1], X[:, 1:2]
        tape.watch(t)
        tape.watch(x)
        u = model(tf.stack([t[:,0], x[:,0]], axis=1))
        u_x = tape.gradient(u, x)
    del tape
    return u_x

def init_model(num_hidden_layers=4, num_neurons_per_layer=80):
    model = tf.keras.Sequential()
    model.add(layers.Input(shape=(2,)))
    # Introduce a scaling layer to map input to [lb, ub]
    # scaling_layer = tf.keras.layers.Lambda(lambda x: (x - lb) / (ub - lb))
    scaling_layer = tf.keras.layers.Lambda(
    lambda x: (1 + 1*(x - lb)/(ub - lb)) / 2  # equivalente a (x-lb)/(ub-lb)
)
    model.add(scaling_layer)
    for _ in range(num_hidden_layers):
        model.add(tf.keras.layers.Dense(num_neurons_per_layer, activation='tanh', kernel_initializer='glorot_normal'))
    model.add(tf.keras.layers.Dense(1))
    return model

model = init_model()



# Define loss and gradient functions
def get_r(model, X_r):
    with tf.GradientTape(persistent=True) as tape:
        t, x = X_r[:, 0:1], X_r[:, 1:2]
        tape.watch(t)
        tape.watch(x)
        u = model(tf.stack([t[:,0], x[:,0]], axis=1))
        u_x = tape.gradient(u, x)
        u_t = tape.gradient(u, t)
    u_xx = tape.gradient(u_x, x)
    del tape
    return u_t - u_xx


def compute_loss(model, X_r, X_data, u_data):
    r = get_r(model, X_r)
    phi_r = tf.reduce_mean(tf.square(r))
    residual_loss = phi_r

    u_pred_ic = model(X_data[0])
    initial_loss = tf.reduce_mean(tf.square(u_pred_ic - u_data[0]))

    u_pred_bc_1 = N_bc(model, X_data[1])
    boundary_loss_N_1 = tf.reduce_mean(tf.square(u_pred_bc_1 - u_data[1]))

    u_pred_bc_2 = N_bc(model, X_data[2])
    boundary_loss_N_2 = tf.reduce_mean(tf.square(u_pred_bc_2 - u_data[2]))

    boundary_loss =  boundary_loss_N_1 +  boundary_loss_N_2

    total_loss = residual_loss + initial_loss + boundary_loss_N_1 + boundary_loss_N_2
    return total_loss, residual_loss, initial_loss, boundary_loss

# Finite differences BC: Neumann-Neumann
#%%
import numpy as np
import matplotlib.pyplot as plt
from scipy.sparse import diags
from scipy.sparse.linalg import spsolve
from matplotlib import cm
# Parámetros de la malla
L = 1
T = 1
Nx = 150  # Número de puntos de la malla en x
Nt = 150  # Número de pasos de tiempo

# Definición de la malla
x = np.linspace(0, L, Nx+1)  # Puntos de la malla en espacio
dx = x[1] - x[0]
t = np.linspace(0, T, Nt+1)  # Puntos de la malla en tiempo
dt = t[1] - t[0]

F = dt / dx**2


def fill_matrix(Nx, F):
    diag_1 = -F * np.ones(Nx)
    diag_0 = (1 + 2 * F) * np.ones(Nx+1)
    diag_2 = -F * np.ones(Nx)

    diagonals = [diag_1, diag_0, diag_2]
    offsets = [-1, 0, 1]

    # Crear la matriz utilizando diags en formato CSR
    A = diags(diagonals, offsets, shape=(Nx + 1, Nx + 1), format='csr')

    # Modificar los elementos específicos para las condiciones de frontera de Neumann
    A[0, 1] = -2 * F
    A[Nx, Nx-1] = -2 * F

    return A

A = fill_matrix(Nx, F)
# print(A)


# Condiciones de frontera de Neumann
f1 = -1
f2 = 0

# Inicializar el vector b_j
bj = np.zeros(Nx+1)
bj[0] = -2 * F * dx * f1
bj[Nx] = 2 * F * dx * f2



def solve_system(A, bj, Nt, Nx):
    # Inicializar la matriz de solución
    u = np.zeros((Nt+1, Nx+1))
    # Condiciones iniciales u(x,0) = 0
    u[0, :] = x*0

    # Bucle temporal
    for j in range(Nt):
        # Resolver el sistema A * u[j+1, :] = u[j, :] + bj
        u[j+1, :] = spsolve(A, u[j, :] + bj)

    return u


# Llamada a la función
u_final = solve_system(A, bj, Nt, Nx)
